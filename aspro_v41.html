function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  // 매핑 시트
  let sh = ss.getSheetByName('aspro_match');
  if (!sh) { sh = ss.insertSheet('aspro_match'); sh.appendRow(['사방넷코드','김포물류품번','메모','등록일']); }
  const data = sh.getDataRange().getValues();
  const map = {};
  data.slice(1).forEach(r => { if (r[0]) map[String(r[0])] = String(r[1]); });
  // 품절 시트
  let so = ss.getSheetByName('aspro_soldout');
  if (!so) { so = ss.insertSheet('aspro_soldout'); so.appendRow(['사방넷코드','등록일']); }
  const soData = so.getDataRange().getValues();
  const soldout = {};
  soData.slice(1).forEach(r => { if (r[0]) soldout[String(r[0])] = true; });
  return ContentService.createTextOutput(JSON.stringify({ok:true, map, soldout}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const body = JSON.parse(e.postData.contents);
  // 매핑 저장
  let sh = ss.getSheetByName('aspro_match');
  if (!sh) { sh = ss.insertSheet('aspro_match'); sh.appendRow(['사방넷코드','김포물류품번','메모','등록일']); }
  const entries = Object.entries(body.map || {});
  const data = sh.getDataRange().getValues();
  const rowIdx = {};
  data.slice(1).forEach((r,i) => { if (r[0]) rowIdx[r[0]] = i+2; });
  let saved = 0;
  entries.forEach(([sc, gp]) => {
    if (rowIdx[sc]) sh.getRange(rowIdx[sc],2).setValue(gp);
    else sh.appendRow([sc, gp, '', new Date().toLocaleDateString('ko-KR')]);
    saved++;
  });
  // 품절 저장
  if (body.soldout) {
    let so = ss.getSheetByName('aspro_soldout');
    if (!so) { so = ss.insertSheet('aspro_soldout'); so.appendRow(['사방넷코드','등록일']); }
    const soEntries = Object.keys(body.soldout);
    const soData = so.getDataRange().getValues();
    const soIdx = {};
    soData.slice(1).forEach((r,i) => { if (r[0]) soIdx[r[0]] = i+2; });
    soEntries.forEach(sc => {
      if (!soIdx[sc]) so.appendRow([sc, new Date().toLocaleDateString('ko-KR')]);
    });
  }
  return ContentService.createTextOutput(JSON.stringify({ok:true, saved}))
    .setMimeType(ContentService.MimeType.JSON);
}
